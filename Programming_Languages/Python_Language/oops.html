What is Object-Oriented Programming?

Object-Oriented Programming (OOP) is a programming paradigm based on the concept of "objects", which can contain data (attributes) and code (methods). OOP focuses on creating reusable, modular code that models real-world entities and relationships.

Analogy: Think of OOP like building with LEGO blocks. Each block (object) has specific properties and can connect with other blocks in defined ways. You can create complex structures by combining these reusable components.

Why Use Object-Oriented Programming?

  • Modularity: Code is organized into self-contained objects
  • Reusability: Classes can be reused across programs
  • Maintainability: Easier to update and maintain code
  • Abstraction: Hide complex implementation details
  • Modeling: Naturally represents real-world entities and relationships

class Car:
    """A simple Car class"""
    
    def __init__(self, make, model, year):
        self.make = make
        self.model = model
        self.year = year
        self.odometer_reading = 0  # Default value

    def read_odometer(self):
        """Print the car's mileage"""
        print(f"This car has {self.odometer_reading} miles on it.")
        
    def update_odometer(self, mileage):
        """Set the odometer reading"""
        if mileage >= self.odometer_reading:
            self.odometer_reading = mileage
        else:
            print("You can't roll back an odometer!")

Inheritance

Inheritance allows a class (child class) to inherit attributes and methods from another class (parent class). This promotes code reuse and establishes relationships between classes.

Inheritance Syntax

class ParentClass:
    # Parent class attributes and methods
    pass

class ChildClass(ParentClass):
    # Child class inherits from ParentClass
    # Can add new attributes and methods
    # Can override parent methods
    pass

class Vehicle:
    """A class representing a vehicle"""
    
    def __init__(self, make, model, year):
        self.make = make
        self.model = model
        self.year = year
        
    def start_engine(self):
        print("Engine started")
        
    def stop_engine(self):
        print("Engine stopped")

class Car(Vehicle):  # Inherits from Vehicle
    """A class representing a car, specialized type of Vehicle"""
    
    def __init__(self, make, model, year, num_doors):
        # Call parent class constructor
        super().__init__(make, model, year)
        self.num_doors = num_doors
        
    def open_trunk(self):
        print("Trunk opened")

class ElectricCar(Car):  # Inherits from Car
    """A class representing an electric car"""
    
    def __init__(self, make, model, year, num_doors, battery_size):
        super().__init__(make, model, year, num_doors)
        self.battery_size = battery_size
        self.charge_level = 100  # Start fully charged
        
    def charge_battery(self, amount):
        """Charge the battery by specified amount"""
        self.charge_level = min(100, self.charge_level + amount)
        print(f"Battery charged to {self.charge_level}%")
        
    # Override parent method
    def start_engine(self):
        print("Electric motor activated")

# Using the classes
my_vehicle = Vehicle("Generic", "Model", 2020)
my_car = Car("Toyota", "Camry", 2022, 4)
my_tesla = ElectricCar("Tesla", "Model 3", 2023, 4, 75)

my_vehicle.start_engine()  # Output: Engine started
my_car.start_engine()      # Output: Engine started (inherited)
my_car.open_trunk()        # Output: Trunk opened
my_tesla.start_engine()    # Output: Electric motor activated (overridden)
my_tesla.charge_battery(20)  # Output: Battery charged to 100%

Encapsulation

Encapsulation is the bundling of data with the methods that operate on that data. It restricts direct access to some of an object's components, which is a way of preventing accidental interference and misuse.

Access Modifiers in Python

Python uses naming conventions to indicate access level rather than strict enforcement:

Access Level Naming Convention Description Example
Public attribute_name Accessible from anywhere self.public_attr
Protected _attribute_name Should not be accessed outside class (convention) self._protected_attr
Private __attribute_name Name mangling makes it harder to access self.__private_attr

Encapsulation Example

class BankAccount:
    """A class representing a bank account with encapsulation"""
    
    def __init__(self, account_holder, initial_balance=0):
        self.account_holder = account_holder  # Public attribute
        self._account_number = self._generate_account_number()  # Protected
        self.__balance = initial_balance  # Private attribute
        
    def _generate_account_number(self):
        """Protected method - for internal use only"""
        import random
        return random.randint(100000, 999999)
        
    def deposit(self, amount):
        """Public method to deposit money"""
        if amount > 0:
            self.__balance += amount
            print(f"Deposited ${amount}. New balance: ${self.__balance}")
        else:
            print("Deposit amount must be positive")
            
    def withdraw(self, amount):
        """Public method to withdraw money"""
        if 0 < amount <= self.__balance:
            self.__balance -= amount
            print(f"Withdrew ${amount}. New balance: ${self.__balance}")
        else:
            print("Invalid withdrawal amount")
            
    def get_balance(self):
        """Public method to get current balance"""
        return self.__balance
        
    # Property for controlled access to account number
    @property
    def account_number(self):
        return f"XXXX{str(self._account_number)[-4:]}"

# Using the BankAccount class
account = BankAccount("John Doe", 1000)

# Accessing public attribute
print(account.account_holder)  # Output: John Doe

# Accessing protected attribute (not recommended but possible)
print(account._account_number)  # Output: 123456 (example)

# Trying to access private attribute directly
# print(account.__balance)  # This would cause an AttributeError

# Using public methods to interact with private data
account.deposit(500)  # Output: Deposited $500. New balance: $1500
account.withdraw(200)  # Output: Withdrew $200. New balance: $1300

# Using property to access masked account number
print(account.account_number)  # Output: XXXX3456

Properties for Controlled Access

Python properties allow you to add getter, setter, and deleter methods to control access to attributes.

class Temperature:
    def __init__(self, celsius=0):
        self._celsius = celsius
        
    @property
    def celsius(self):
        """Getter for celsius"""
        return self._celsius
        
    @celsius.setter
    def celsius(self, value):
        """Setter for celsius with validation"""
        if value < -273.15:
            raise ValueError("Temperature below absolute zero is not possible")
        self._celsius = value
        
    @property
    def fahrenheit(self):
        """Read-only property for fahrenheit"""
        return (self._celsius * 9/5) + 32

# Using the Temperature class
temp = Temperature(25)
print(temp.celsius)     # Output: 25
print(temp.fahrenheit)  # Output: 77.0

temp.celsius = 30
print(temp.celsius)     # Output: 30
print(temp.fahrenheit)  # Output: 86.0

# temp.celsius = -300  # Would raise ValueError

Polymorphism

Polymorphism allows objects of different classes to be treated as objects of a common super class. It enables using a single interface to represent different underlying forms (data types).

Polymorphism Through Inheritance

class Animal:
    def speak(self):
        raise NotImplementedError("Subclass must implement this method")

class Dog(Animal):
    def speak(self):
        return "Woof!"

class Cat(Animal):
    def speak(self):
        return "Meow!"

class Duck(Animal):
    def speak(self):
        return "Quack!"

# Polymorphic function
def animal_sound(animal):
    print(animal.speak())

# Create different animal objects
animals = [Dog(), Cat(), Duck()]

# Each animal responds differently to the same method call
for animal in animals:
    animal_sound(animal)
# Output:
# Woof!
# Meow!
# Quack!

Polymorphism With Duck Typing

Python uses "duck typing" - if an object walks like a duck and quacks like a duck, it's treated as a duck, regardless of its actual type.

class TextFile:
    def read(self):
        return "Text content from file"

class CSVFile:
    def read(self):
        return "CSV data from file"

class Database:
    def read(self):
        return "Data from database"

# This function doesn't care about the object's type
# It only cares that the object has a read() method
def process_data(data_source):
    print(f"Processing: {data_source.read()}")

# Different objects with the same method
sources = [TextFile(), CSVFile(), Database()]

for source in sources:
    process_data(source)
# Output:
# Processing: Text content from file
# Processing: CSV data from file
# Processing: Data from database

Operator Overloading

Python allows operators to have different meanings according to context (operator overloading).

class Vector:
    def __init__(self, x, y):
        self.x = x
        self.y = y
        
    def __add__(self, other):
        """Overload the + operator"""
        return Vector(self.x + other.x, self.y + other.y)
        
    def __sub__(self, other):
        """Overload the - operator"""
        return Vector(self.x - other.x, self.y - other.y)
        
    def __mul__(self, scalar):
        """Overload the * operator for scalar multiplication"""
        return Vector(self.x * scalar, self.y * scalar)
        
    def __str__(self):
        """Overload the str() function"""
        return f"Vector({self.x}, {self.y})"

# Using operator overloading
v1 = Vector(2, 3)
v2 = Vector(1, 2)

v3 = v1 + v2  # Uses __add__
print(v3)  # Output: Vector(3, 5)

v4 = v1 - v2  # Uses __sub__
print(v4)  # Output: Vector(1, 1)

v5 = v1 * 3  # Uses __mul__
print(v5)  # Output: Vector(6, 9)

Magic Methods

Magic methods (also called dunder methods) are special methods that start and end with double underscores. They allow you to define how objects of your class behave with Python's built-in operations.

Common Magic Methods

Method Purpose Example
__init__ Object initialization obj = MyClass()
__str__ String representation print(obj)
__len__ Length of object len(obj)
__getitem__ Indexing obj[index]
__setitem__ Assignment by index obj[index] = value
__add__ Addition obj1 + obj2
__eq__ Equality comparison obj1 == obj2

Magic Methods Example

class Book:
    def __init__(self, title, author, pages):
        self.title = title
        self.author = author
        self.pages = pages
        
    def __str__(self):
        return f"'{self.title}' by {self.author}"
        
    def __len__(self):
        return self.pages
        
    def __eq__(self, other):
        if isinstance(other, Book):
            return (self.title == other.title and 
                    self.author == other.author)
        return False
        
    def __add__(self, other):
        if isinstance(other, Book):
            # Create a bookshelf with both books
            return Bookshelf([self, other])
        return NotImplemented
        
    def __getitem__(self, key):
        # Allow book['title'] or book['author']
        if key == 'title':
            return self.title
        elif key == 'author':
            return self.author
        elif key == 'pages':
            return self.pages
        else:
            raise KeyError(f"Book has no attribute '{key}'")

class Bookshelf:
    def __init__(self, books=None):
        self.books = books or []
        
    def __len__(self):
        return len(self.books)
        
    def __getitem__(self, index):
        return self.books[index]
        
    def __str__(self):
        return f"Bookshelf with {len(self)} books"

# Using the Book class
book1 = Book("Python Basics", "John Doe", 300)
book2 = Book("Advanced Python", "Jane Smith", 450)

print(book1)  # Output: 'Python Basics' by John Doe (uses __str__)
print(len(book1))  # Output: 300 (uses __len__)

# Comparison
book3 = Book("Python Basics", "John Doe", 350)
print(book1 == book3)  # Output: True (uses __eq__)
print(book1 == book2)  # Output: False

# Addition creates a bookshelf
shelf = book1 + book2  # Uses __add__
print(shelf)  # Output: Bookshelf with 2 books
print(len(shelf))  # Output: 2
print(shelf[0])  # Output: 'Python Basics' by John Doe

# Dictionary-like access
print(book1['title'])  # Output: Python Basics (uses __getitem__)
print(book1['author'])  # Output: John Doe

class Product:

        self.price = price
        self.quantity = quantity

        self.quantity += amount

class ShoppingCart:
    def __init__(self):
        self.items = {}
        
    def add_item(self, product, quantity=1):
        if product.quantity >= quantity:
            if product in self.items:
                self.items[product] += quantity
            else:
                self.items[product] = quantity
            product.quantity -= quantity
        else:
            print("Insufficient stock")
            
    def get_total(self):
        return sum(product.price * qty for product, qty in self.items.items())

Best Practices

Follow these best practices to write clean, maintainable, and efficient object-oriented code.

1. Follow the Single Responsibility Principle

# Bad - class does too many things
class UserManager:
    def __init__(self, username, email):
        self.username = username
        self.email = email
        
    def save_to_database(self):
        # database logic
        pass
        
    def send_email(self, message):
        # email logic
        pass
        
    def validate_data(self):
        # validation logic
        pass

# Better - separate concerns
class User:
    def __init__(self, username, email):
        self.username = username
        self.email = email
        
class UserValidator:
    @staticmethod
    def validate(user):
        # validation logic
        pass
        
class UserRepository:
    def save(self, user):
        # database logic
        pass
        
class EmailService:
    def send(self, user, message):
        # email logic
        pass

2. Use Composition Over Inheritance

# Inheritance can lead to complex hierarchies
class Vehicle:
    pass

class Car(Vehicle):
    pass

class ElectricCar(Car):
    pass

class HybridCar(ElectricCar):
    pass  # Gets complicated!

# Composition is often more flexible
class Engine:
    pass

class Battery:
    pass

class Car:
    def __init__(self):
        self.engine = Engine()
        self.battery = Battery()  # Optional

3. Prefer Private Attributes with Getters/Setters

# Direct access allows invalid states
class Person:
    def __init__(self, age):
        self.age = age  # Could be set to negative

person = Person(-5)  # Invalid age

# Use properties for validation
class Person:
    def __init__(self, age):
        self._age = age
        
    @property
    def age(self):
        return self._age
        
    @age.setter
    def age(self, value):
        if value < 0:
            raise ValueError("Age cannot be negative")
        self._age = value

person = Person(25)
person.age = 30  # OK
# person.age = -5  # Raises ValueError

5. Write Comprehensive Docstrings

class BankAccount:
